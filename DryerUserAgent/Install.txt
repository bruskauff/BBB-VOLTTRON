#Instructions for installing the Dryer User Agent into VOLTTRON

#First copy the entire DryerUserAgent folder into volttron/Agents

#Copy and paste the code below into the lx command line of the BeagleBone Black (BBB).
#This assumes the BBB was just turned on.

cd volttron
. env/bin/activate
volttron -l volttron.log -vv&
volttron-pkg package Agents/DryerUserAgent
volttron-pkg configure /tmp/volttron_wheels/dryeruseragent-0.1-py2-none-any.whl Agents/DryerUserAgent/dryeruseragent.config
volttron-ctl install /tmp/volttron_wheels/dryeruseragent-0.1-py2-none-any.whl

#Check the status of the install with
volttron-ctl status
